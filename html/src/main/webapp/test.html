const express = require('express'); const path = require('path'); const bodyParser = require('body-parser'); const app = express(); app.use(express.static(path.join(__dirname, 'public'))); app.use(bodyParser.urlencoded({ extended: false })); app.post('/post', (req, res) => { res.send(req.body); }); app.listen(3000); console.log('服务器启动成功');
